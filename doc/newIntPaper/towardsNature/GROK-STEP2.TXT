def compute_delta_D(cm, dynamic, params, knockout_node):
    # Original D
    D_orig = compute_D(cm, dynamic, params)
    # Knockout: Set row/column to 0, recompute
    cm_ko = cm.copy()
    cm_ko[knockout_node, :] = 0
    cm_ko[:, knockout_node] = 0
    D_ko = compute_D(cm_ko, dynamic, params)
