StatisticalComparison[bioD_, randomDs_] := Module[
  {tTest, permTest, mannWhitney, effectSize},
  
  (* 1. Parametric: One-tailed t-test *)
  tTest = TTest[randomDs, bioD, "Less"];

  permTest = PermutationTest[
